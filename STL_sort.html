sort()

  • C++의 algorithm 라이브러리에는 sort()함수가 정의되어있다.
  • sort()함수가 강력한 이유는 정렬 할 기준을 마음대로 정의할 수 있기 때문이다. => sort함수 안에 매개변수로 넣어준다.

    • 예시
     bool desc_compare(int a, int b) {
          return a > b; //a가 b보다 클 때 우선적으로 정렬, 내림차순 정렬
     }

데이터를 묶어서 정렬

  • 프로그래밍 할 때는 모든 데이터들이 객체로 정리되어 내부적으로 여러 개의 변수를 포함하고 있다.
  • '특정한 변수를 기준으로' 정렬하는 것이 중요하다.
  • -클래스를 정의하는 방식은 프로그래밍 속도 측면에서 별로 유리하지 않다.

페어(Pair)라이브러리 사용

  • 프로그래밍 대회와 같이 빠른 개발이 필요할 때는 Pair(페어) 라이브러리를 사용하는 것이 더 효율적이다.
  • 벡터(Vector) 라이브러리와 페어(Pair) 라이브러리를 이용해 배열과 클래스 이용 방식을 대체
  • 벡터 STL은 원소를 선택적으로 삽입(Push) 및 삭제(Pop) 할 수 있다.
  • 페어(Pair) STL은 한 쌍의 데이터를 처리할 수 있도록 해주는 자료구조이다.

#include<iostream>
#include<string>
#include<vector> // 연결리스트 형태로 표현으로 더 효율적이다.
#include<algorithm>
 
using namespace std;
/* 학생을 나타낼 수 있는 정보가 이름,성적 ,생년월일일 떄 학생을 성적 순서대로 나열하고자 한다. 
    다만 성적이 동일한 경우 나이가 더 어린 학생이 더 우선순위가 높다.*/
 
bool compare(pair<string, pair<int, int>>a,
    pair<string, pair<int, int>>b) {
    if (a.second.first == b.second.first) {//성적이 동일하다면
        return a.second.second > a.second.second; // 생년월일이 더 늦은 사람이 우선순위를 가지도록 한다.
    }
    else { // 성적이 다르다면 성적 순으로 나열한다.
        return a.second.first > b.second.first;
 
    }
    }
int main(void) {
    vector<pair<string,pair<int, int>>>v; // 이중페어
    v.push_back(pair<string, pair<int, int>>("나동빈",pair<int, int>(90,19961222)));
    v.push_back(pair<string, pair<int, int>>("이태일", pair<int, int>(97, 19930518)));
    v.push_back(pair<string, pair<int, int>>("박한올", pair<int, int>(95, 19930203)));
    v.push_back(pair<string, pair<int, int>>("이상욱", pair<int, int>(90, 19921207)));
    v.push_back(pair<string, pair<int, int>>("강종구", pair<int, int>(88, 19900302)));
 
 
    sort(v.begin(), v.end(),compare); //정렬 기준을 알려줘야 한다.
    for (int i = 0; i < v.size(); i++) //size()는 벡터의 크기를 가져오는 func
    {
        cout << v[i].first << ' '; 
    }
 
    return 0;
}
